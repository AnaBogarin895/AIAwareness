                    <h3 class="text-2xl font-bold text-red-600 mb-6 flex items-center">
                        <i class="fas fa-times-circle mr-3"></i>
                        Don't Do This
                    </h3>


                    <div class="space-y-4">
                        <div class="border-l-4 border-red-500 pl-4 py-2">
                            <p class="font-semibold text-red-700">❌ "Help me with this"</p>
                            <p class="text-gray-600 text-sm">Too vague - AI doesn't know what "this" is or how to help</p>
                        </div>


                        <div class="border-l-4 border-red-500 pl-4 py-2">
                            <p class="font-semibold text-red-700">❌ "Write something about marketing"</p>
                            <p class="text-gray-600 text-sm">No specific task, format, or outcome defined</p>
                        </div>


                        <div class="border-l-4 border-red-500 pl-4 py-2">
                            <p class="font-semibold text-red-700">❌ "Make it better"</p>
                            <p class="text-gray-600 text-sm">"Better" is subjective - specify what improvements you want</p>
                        </div>


                        <div class="border-l-4 border-red-500 pl-4 py-2">
                            <p class="font-semibold text-red-700">❌ Single long paragraph with multiple requests</p>
                            <p class="text-gray-600 text-sm">Confusing and hard to process - break into clear sections</p>
                        </div>
                    </div>
                </div>


                <div class="bg-white p-8 rounded-xl shadow-lg">
                    <h3 class="text-2xl font-bold text-green-600 mb-6 flex items-center">

                        Do This Instead
                    </h3>


                    <div class="space-y-4">
                        <div class="border-l-4 border-green-500 pl-4 py-2">
                            <p class="font-semibold text-green-700">✅ "You are a marketing expert. Analyze this email campaign's open rates and suggest 3 specific improvements."</p>
                            <p class="text-gray-600 text-sm">Clear role, task, and expected output</p>
                        </div>


                        <div class="border-l-4 border-green-500 pl-4 py-2">
                            <p class="font-semibold text-green-700">✅ "Create a 5-slide presentation outline for our Q4 product launch, targeting enterprise clients."</p>

                            <p class="font-semibold text-green-700">✅ "Make this email more persuasive by adding social proof and urgency. Keep it under 200 words."</p>
                            <p class="text-gray-600 text-sm">Specific improvements and constraints provided</p>
                        </div>

                            <p class="font-semibold text-green-700">✅ Use RISEN or CARE frameworks</p>

        function analyzePrompt() {
            const prompt = document.getElementById('promptInput').value;
            const analysisDiv = document.getElementById('promptAnalysis');
            const qualityScore = document.getElementById('qualityScore');
            const qualityBar = document.getElementById('qualityBar');


            if (!prompt.trim()) {
                analysisDiv.innerHTML = `
                    <div class="text-center text-gray-400 py-8">
                        <i class="fas fa-magic text-3xl mb-4 opacity-50"></i>
                        <p>Start typing your prompt to see real-time analysis and suggestions!</p>
                    </div>
                `;
                qualityScore.textContent = '0/100';
                qualityBar.style.width = '0%';
                return;
            }


            // Simple prompt analysis
            let score = 0;
            const analysis = [];


            // Check for role definition
            if (prompt.toLowerCase().includes('you are') || prompt.toLowerCase().includes('act as')) {
                score += 20;
                analysis.push('<div class="flex items-center text-green-400 mb-2"><i class="fas fa-check mr-2"></i>Role defined</div>');
            } else {
                analysis.push('<div class="flex items-center text-yellow-400 mb-2"><i class="fas fa-exclamation mr-2"></i>Consider adding a role (e.g., "You are a...")</div>');
            }


            // Check for specificity
            if (prompt.length > 50) {
                score += 20;
                analysis.push('<div class="flex items-center text-green-400 mb-2"><i class="fas fa-check mr-2"></i>Good detail level</div>');
            } else {
                analysis.push('<div class="flex items-center text-yellow-400 mb-2"><i class="fas fa-exclamation mr-2"></i>Add more specific details</div>');
            }


            // Check for format specification
            if (prompt.toLowerCase().includes('format') || prompt.toLowerCase().includes('table') || prompt.toLowerCase().includes('list')) {
                score += 20;
                analysis.push('<div class="flex items-center text-green-400 mb-2"><i class="fas fa-check mr-2"></i>Output format specified</div>');
            } else {
                analysis.push('<div class="flex items-center text-yellow-400 mb-2"><i class="fas fa-exclamation mr-2"></i>Specify output format</div>');
            }


            // Check for constraints
            if (prompt.toLowerCase().includes('words') || prompt.toLowerCase().includes('bullets') || prompt.toLowerCase().includes('sentences')) {
                score += 20;
                analysis.push('<div class="flex items-center text-green-400 mb-2"><i class="fas fa-check mr-2"></i>Constraints provided</div>');
            } else {
                analysis.push('<div class="flex items-center text-yellow-400 mb-2"><i class="fas fa-exclamation mr-2"></i>Consider adding length/format constraints</div>');
            }


            // Check for action words
            const actionWords = ['analyze', 'create', 'write', 'generate', 'summarize', 'explain', 'compare'];
            const hasActionWord = actionWords.some(word => prompt.toLowerCase().includes(word));
            if (hasActionWord) {
                score += 20;
                analysis.push('<div class="flex items-center text-green-400 mb-2"><i class="fas fa-check mr-2"></i>Clear action specified</div>');
            } else {
                analysis.push('<div class="flex items-center text-yellow-400 mb-2"><i class="fas fa-exclamation mr-2"></i>Use clear action words (analyze, create, write, etc.)</div>');
            }


            analysisDiv.innerHTML = `
                <h4 class="font-bold text-white mb-4">Prompt Analysis</h4>
                ${analysis.join('')}
                <div class="mt-4 p-3 bg-blue-500 bg-opacity-20 rounded-lg">
                    <p class="text-sm text-blue-200">
                        <strong>Tip:</strong> ${getRandomTip()}
                    </p>
                </div>
            `;


            qualityScore.textContent = `${score}/100`;
            qualityBar.style.width = `${score}%`;
        }

        // Save Prompt Functionality
        function savePrompt() {
            const prompt = document.getElementById('promptInput').value;
            if (!prompt.trim()) {
                alert('Please enter a prompt to save.');
                return;
            }


            const savedPrompts = JSON.parse(localStorage.getItem('savedPrompts') || '[]');
            const newPrompt = {
                id: Date.now(),
                content: prompt,
                timestamp: new Date().toLocaleDateString(),
                title: prompt.substring(0, 50) + (prompt.length > 50 ? '...' : '')
            };


            savedPrompts.unshift(newPrompt);
            localStorage.setItem('savedPrompts', JSON.stringify(savedPrompts.slice(0, 10))); // Keep only 10 most recent


            displaySavedPrompts();
            alert('Prompt saved successfully!');
        }


        function displaySavedPrompts() {
            const savedPrompts = JSON.parse(localStorage.getItem('savedPrompts') || '[]');
            const container = document.getElementById('savedPrompts');


            if (savedPrompts.length === 0) {
                container.innerHTML = '<div class="col-span-full text-center text-gray-400 py-8">No saved prompts yet. Save some prompts to see them here!</div>';
                return;
            }


            container.innerHTML = savedPrompts.map(prompt => `
                <div class="bg-white bg-opacity-10 p-4 rounded-lg">
                    <h4 class="font-bold text-white mb-2">${prompt.title}</h4>
                    <p class="text-gray-300 text-sm mb-3 line-clamp-3">${prompt.content}</p>
                    <div class="flex justify-between items-center">
                        <span class="text-xs text-gray-400">${prompt.timestamp}</span>
                        <div class="space-x-2">
                            <button onclick="loadPrompt(${prompt.id})" class="px-3 py-1 text-xs bg-blue-600 text-white rounded hover:bg-blue-700">Load</button>
                            <button onclick="deletePrompt(${prompt.id})" class="px-3 py-1 text-xs bg-red-600 text-white rounded hover:bg-red-700">Delete</button>
                        </div>
                    </div>
                </div>
            `).join('');
        }


        function loadPrompt(id) {
            const savedPrompts = JSON.parse(localStorage.getItem('savedPrompts') || '[]');
            const prompt = savedPrompts.find(p => p.id === id);
            if (prompt) {
                document.getElementById('promptInput').value = prompt.content;
                analyzePrompt();
                // Scroll to playground
                document.getElementById('promptInput').scrollIntoView({
                    behavior: 'smooth'
                });
            }
        }


        function deletePrompt(id) {
            if (confirm('Are you sure you want to delete this prompt?')) {
                const savedPrompts = JSON.parse(localStorage.getItem('savedPrompts') || '[]');
                const filteredPrompts = savedPrompts.filter(p => p.id !== id);
                localStorage.setItem('savedPrompts', JSON.stringify(filteredPrompts));
                displaySavedPrompts();
            }
        }


        // Initialize saved prompts on page load
        document.addEventListener('DOMContentLoaded', function() {
            displaySavedPrompts();


            // Add some animation delays
            const elements = document.querySelectorAll('.fade-in-up');
            elements.forEach((el, index) => {
                el.style.animationDelay = `${index * 0.1}s`;
            });
        });


        // Add smooth scrolling for internal links
        document.addEventListener('click', function(e) {
            if (e.target.tagName === 'A' && e.target.getAttribute('href').startsWith('#')) {
                e.preventDefault();
                const target = document.querySelector(e.target.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth'
                    });
                }
            }
        });
